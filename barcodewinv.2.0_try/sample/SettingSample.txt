[Settings]

' 入力フォルダ（絶対パス）
InputPath = .\sample\image
	' 特定フォルダ内のファイルを一括処理する場合に用いる
	' 未指定時は一括処理を行わない

' 画像ファイル形式
InputExt = tif|bmp|jpg|png|pdf
	' 特定フォルダ内のファイルを一括処理する際のファイル拡張子フィルタ
	' 区切り文字「|」で複数指定可
	' tif TIFF（既定値）
	' pdf PDF
	' jpg JPEG
	' png PNG
	' bmp Bitmap

' 単位
'Unit = 1
	' 処理矩形の単位 0=ピクセル(画素数)、1=ミリ(0.1mm 単位)
	' 未指定時：0
	' ※処理矩形を使わない場合は無視
	' PDF から領域指定で読み取る場合、または、PDF から読み取り座標を取得する場合は、
	' ミリ単位(1)を指定ください。

' 解像度
'DpiX = 400
'DpiY = 400
	' 0を指定、または指定しなければ画像ファイル内の解像度情報に従う

' 頁指定
RecogPage = 0
	' 認識する頁番号： =0:全頁、>0:指定頁
	' 存在しないページ番号の場合処理しない
	' 未指定時：1
	' マイナスや文字列しかない場合は未指定

' 最大検出数
Max = 10
	' １頁内のQR ｺｰﾄﾞ検出最大数を指定（既定値：10）。

' 処理矩形
'Area = 0|0|180|200
	' 認識エリア指定 x|y|w|h （値は4 つ必要）
	' 未指定時： 0|0|0|0 (自動検出)
	' 解析失敗時： 0|0|0|0 (自動検出)
	' PDF から読み取り座標を取得する場合は処理矩形の単位をミリ単位としてください。

' 位置検出レベル
'Finder= 0
	' -3: 条件緩和・強 .... 条件を緩和して位置検出ﾊﾟﾀｰﾝを検索
	' -2: 条件緩和・中 ：
	' -1: 条件緩和・弱 ：
	' 0: 標準（既定値）....通常の検索
	' 1: 拡大検出・弱 ：
	' 2: 拡大検出・中 ：
	' 3: 拡大検出・強 ....画像を拡大して位置検出ﾊﾟﾀｰﾝを検索

' 構造的連接設定
'StrApp= 0
	' 構造的連接設定に対応したQR コードを結合します。
	' 0: 結合なし（初期値）
	' 1: 同一ページ内で結合する。ページをまたぐQR の分割は結合しない。
	' 2: 同一ファイル内で結合する(ページをまたいだQRの分割も結合可能)。
	'    異なるファイルのQR の分割は結合しない。
	' ※1, 2 を指定した場合、
	' 　構造的連接に対応した全てのQR コードの読み取りに成功した場合のみ結果を返します。
	' 　構造的連接に対応した結合前の個々のQR コードを結果として返すことはありません。
	' 　StrApp=1 の場合、RecogPage=0 または、任意のページを指定してください。
	' 　StrApp=2 の場合、RecogPage=0 を指定してください。




' 出力ファイル
OutputFile = .\csvout.txt
	' 出力テキストのファイルパス
	' 未指定時：標準出力のみ

' 上書き
OverWrite = 1
	' 上書きか追記か 0=追記,1=上書き
	' 未指定時：追記
	' 範囲外や文字列しかない場合は未指定

' 出力書式
OutputFormat = %Data, %RetVal, %InputPath%CRLF
	' %RetVal, %Data, %InputPath
	' 出力テキストの内容（「出力書式」の項を参照）
	' ※書式指定文字以外の文字列はそのまま出力
	' 未指定時："%Data%CRLF" （認識結果＋改行）



' [EOF]
